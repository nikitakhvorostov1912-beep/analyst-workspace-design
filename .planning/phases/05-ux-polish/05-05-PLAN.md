```
test.describe("First-run onboarding", () => {
  test.beforeEach(async ({ page }) => {
    await setupOnboardingMocks(page, { initialConnections: [], initialLLM: null });
    await page.addInitScript(() => { localStorage.clear(); sessionStorage.clear(); });
  });

  test("показывает onboarding на пустой БД и проводит через 3 шага", async ({ page }) => {
    const start = Date.now();
    await page.goto("/");
    // Шаг 1
    await expect(page.getByText(/Подключите вашу базу 1С/i)).toBeVisible();
    await page.getByLabel(/Название/i).fill("Local 1C");
    await page.getByLabel(/Endpoint/i).fill("http://localhost:6010/mcp");
    await page.getByRole("button", { name: /Сохранить/i }).click();
    await expect(page.getByText(/MCP подключён/i)).toBeVisible();
    await page.getByRole("button", { name: /Далее/i }).click();
    // Шаг 2
    await expect(page.getByText(/Настройте LLM/i)).toBeVisible();
    await page.getByLabel(/Endpoint/i).fill("http://localhost:1234/v1");
    await page.getByLabel(/Модель/i).fill("gpt-4o-mini");
    await page.getByLabel(/API ключ/i).fill("sk-test-1234567890");
    await page.getByRole("button", { name: /Тест/i }).click();
    await expect(page.getByText(/LLM работает/i)).toBeVisible();
    await page.getByRole("button", { name: /Сохранить/i }).click();
    await page.getByRole("button", { name: /Далее/i }).click();
    // Шаг 3
    await expect(page.getByText(/Готово/i)).toBeVisible();
    await page.getByRole("button", { name: /Начать работу/i }).click();
    // Модалка закрылась
    await expect(page.getByText(/Подключите вашу базу 1С/i)).not.toBeVisible();
    const elapsed = (Date.now() - start) / 1000;
    expect(elapsed).toBeLessThan(90);
  });

  test("Пропустить закрывает модалку и ставит флаг", async ({ page }) => {
    await page.goto("/");
    await expect(page.getByText(/Подключите вашу базу 1С/i)).toBeVisible();
    await page.getByRole("button", { name: /Пропустить/i }).click();
    await expect(page.getByText(/Подключите вашу базу 1С/i)).not.toBeVisible();
    const flag = await page.evaluate(() => localStorage.getItem("analyst.onboarding_completed"));
    expect(flag).toBe("true");
  });

  test("legacy: backend имеет данные → onboarding не показывается", async ({ page }) => {
    await setupOnboardingMocks(page, {
      initialConnections: [{ id: "c1", name: "Legacy", endpoint: "http://x", channel: null, anon_enabled: false, last_seen_at: null, created_at: new Date().toISOString() }],
      initialLLM: { id: "default", endpoint: "http://x", model: "y", temperature: 0.3 }
    });
    await page.addInitScript(() => { localStorage.clear(); });
    await page.goto("/");
    await page.waitForLoadState("networkidle");
    await expect(page.getByText(/Подключите вашу базу 1С/i)).not.toBeVisible();
  });

  test("повторный refresh → onboarding не открывается", async ({ page, context }) => {
    await context.addInitScript(() => localStorage.setItem("analyst.onboarding_completed", "true"));
    await page.goto("/");
    await expect(page.getByText(/Подключите вашу базу 1С/i)).not.toBeVisible();
  });
});
```
