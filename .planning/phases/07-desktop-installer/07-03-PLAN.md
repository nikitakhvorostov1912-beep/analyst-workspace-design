<task type="auto">
  <name>Task 2: Создать build-bundle.js — pipeline сборки frontend → resources/frontend/</name>
  <files>desktop/scripts/build-bundle.js</files>
  <action>
    Создать `desktop/scripts/build-bundle.js`. CommonJS (require/module.exports — НЕ ESM, чтобы запускать через `node` без type=module ограничений в package.json). Скрипт ≤100 строк.

    Структура (см. 07-CONTEXT.md секция «Build pipeline»):
    ```js
    const { execSync } = require('child_process');
    const fs = require('fs');
    const path = require('path');

    const DESKTOP = path.resolve(__dirname, '..');
    const ROOT = path.resolve(DESKTOP, '..');
    const RES = path.join(DESKTOP, 'resources');

    // Args parsing: --skip-backend (для wave 1 когда backend.exe ещё не собран)
    const args = process.argv.slice(2);
    const skipBackend = args.includes('--skip-backend');
    const skipFrontend = args.includes('--skip-frontend');

    fs.mkdirSync(RES, { recursive: true });

    // === 1. Backend (PyInstaller) ===
    if (!skipBackend) {
      console.log('[1/3] Building backend.exe via PyInstaller...');
      execSync('pyinstaller --clean --noconfirm build.spec', {
        cwd: path.join(ROOT, 'backend'),
        stdio: 'inherit',
      });
      const src = path.join(ROOT, 'backend', 'dist', 'backend.exe');
      const dst = path.join(RES, 'backend.exe');
      if (!fs.existsSync(src)) throw new Error('backend.exe not produced');
      fs.copyFileSync(src, dst);
      console.log(`    → ${dst} (${(fs.statSync(dst).size / 1024 / 1024).toFixed(1)} MB)`);
    } else {
      console.log('[1/3] Backend skipped');
    }

    // === 2. Next standalone ===
    if (!skipFrontend) {
      console.log('[2/3] Building Next.js standalone...');
      execSync('pnpm build', {
        cwd: path.join(ROOT, 'frontend'),
        env: { ...process.env, NEXT_OUTPUT: 'standalone' },
        stdio: 'inherit',
        shell: true,
      });

      const STANDALONE_SRC = path.join(ROOT, 'frontend', '.next', 'standalone', 'frontend');
      const NM_SRC = path.join(ROOT, 'frontend', '.next', 'standalone', 'node_modules');
      const STATIC_SRC = path.join(ROOT, 'frontend', '.next', 'static');
      const PUBLIC_SRC = path.join(ROOT, 'frontend', 'public');
      const FE_DEST = path.join(RES, 'frontend');

      console.log('[3/3] Copying standalone → resources/frontend/...');
      fs.rmSync(FE_DEST, { recursive: true, force: true });
      fs.mkdirSync(FE_DEST, { recursive: true });

      // Windows caveat: NEVER use symlinks; dereference: true резолвит links в файлы
      const copyOpts = { recursive: true, dereference: true };
      fs.cpSync(STANDALONE_SRC, FE_DEST, copyOpts);
      fs.cpSync(NM_SRC, path.join(FE_DEST, 'node_modules'), copyOpts);
      fs.cpSync(STATIC_SRC, path.join(FE_DEST, '.next', 'static'), copyOpts);
      if (fs.existsSync(PUBLIC_SRC)) {
        fs.cpSync(PUBLIC_SRC, path.join(FE_DEST, 'public'), copyOpts);
      }

      // Verify: server.js must exist
      const serverJs = path.join(FE_DEST, 'server.js');
      if (!fs.existsSync(serverJs)) {
        throw new Error(`server.js not found at ${serverJs} after copy`);
      }
      console.log(`    → ${FE_DEST}/server.js`);
    } else {
      console.log('[2/3 & 3/3] Frontend skipped');
    }

    console.log('Bundle ready. Run: pnpm dlx electron-builder --win');
    ```

    **Каверзы (учтены):**
    - `dereference: true` — копирует резолвленные symlinks как файлы (Windows без admin не умеет symlinks). Это критично для node_modules где Next.js может использовать ссылки.
    - `shell: true` для execSync на Windows — pnpm.cmd иначе не находится.
    - Throw on missing server.js — fast failure если standalone не собрался.

    Karpathy: один файл, последовательный happy-path, никаких promise chains, никакого parallel execution (PyInstaller и pnpm build могут конфликтовать за CPU — последовательно проще).
  </action>
