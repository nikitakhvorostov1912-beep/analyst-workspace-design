```ts
export function formatDuration(ms: number | null | undefined): string {
  if (ms == null || ms < 0) return "";
  if (ms < 1000) return `${ms} мс`;
  if (ms < 60_000) return `${(ms / 1000).toFixed(1)} с`;
  const m = Math.floor(ms / 60_000);
  const s = Math.floor((ms % 60_000) / 1000);
  return `${m} мин ${s} с`;
}
```

1. **ToolTrace.tsx:**
```tsx
"use client";
import { useState } from "react";
import { JsonTree } from "@/lib/json-tree";
import { formatDuration } from "@/lib/format-duration";
import { ChevronRight, ChevronDown } from "lucide-react";
import type { ToolCallRecord } from "@/lib/types";

type ToolTraceProps = {
  toolCalls: ToolCallRecord[];
  totalDurationMs?: number;
};

export function ToolTrace({ toolCalls, totalDurationMs }: ToolTraceProps) {
  const [open, setOpen] = useState(false);

  if (!toolCalls || toolCalls.length === 0) return null;

  return (
    <div className="mt-2 text-xs text-[var(--fg-muted)]">
      <button
        type="button"
        onClick={() => setOpen(!open)}
        className="flex items-center gap-1 hover:text-[var(--fg)] transition-colors"
        aria-expanded={open}
      >
        {open ? <ChevronDown size={12} /> : <ChevronRight size={12} />}
        <span>
          LLM вызвала {toolCalls.length} tool{toolCalls.length > 1 ? 'a' : ''}
          {totalDurationMs ? `, ${formatDuration(totalDurationMs)}` : ''}
        </span>
      </button>

      {open && (
        <ul className="mt-2 space-y-3 border-l-2 border-[var(--border)] pl-3">
          {toolCalls.map((tc) => (
            <li key={tc.id} className="space-y-1">
              <div className="flex items-baseline gap-2">
                <span className="font-mono text-[var(--fg)]">{tc.name}</span>
                {tc.duration_ms ? <span className="text-[var(--fg-muted)]">· {formatDuration(tc.duration_ms)}</span> : null}
                {tc.ok === false ? <span className="text-red-400">· ошибка</span> : null}
              </div>

              <details className="ml-2">
                <summary className="cursor-pointer text-[var(--fg-muted)] hover:text-[var(--fg)]">args</summary>
                <div className="mt-1 ml-2">
                  <JsonTree value={tc.args} defaultExpanded={1} />
                </div>
              </details>

              {tc.result !== undefined && (
                <details className="ml-2">
                  <summary className="cursor-pointer text-[var(--fg-muted)] hover:text-[var(--fg)]">result</summary>
                  <div className="mt-1 ml-2">
                    <JsonTree value={tc.result} defaultExpanded={0} />
                  </div>
                </details>
              )}

              {tc.error ? (
                <div className="ml-2 text-red-400">
                  <span className="font-mono">{tc.error}</span>
                </div>
              ) : null}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

<risks>
1. **Огромный result JSON (10К-50К символов) фризит JsonTree** — рекурсивный рендер каждого узла = много DOM-нод. **Митигация:** result в JsonTree рендерится только при click (lazy через `<details>` — не монтируется до open). А внутри JsonTree обрезаем строки до 200 chars. Если объект массивный (>500 keys) — обрезаем массивы до 100 элементов с «...ещё N скрыто» (опционально, можно отложить).
2. **JSON.stringify для cycle detection — slow на больших объектах** — каждый JsonTree вызов делает stringify целиком. **Митигация:** делаем stringify только в catch-блоке когда детектим цикл; primary рендеринг — рекурсия через typeof. Если массивный объект → стримим по чанку (typeof === 'object' && !Array.isArray → Object.keys; этого достаточно без stringify).
3. **defaultExpanded depth 0 — пользователь не увидит args сразу** — UX слегка хуже. **Митигация:** args defaultExpanded=1 (видны ключи верхнего уровня), result defaultExpanded=0 (свёрнут, открывается по click).
4. **Live стриминг tool_calls — частичный объект args (Plan 2.3 useChatStream)** — пока args накапливается, render может падать на невалидном JSON. **Митигация:** useChatStream обновляет args только когда event:tool_call полностью получен (там args уже dict из backend, не строка). А tool_result обновляет result когда event:tool_result пришёл. Промежуточных состояний нет.
5. **ToolTrace не показывает status events (thinking/calling_tool/formatting)** — пользователь не видит streaming stages. **Митигация:** это явное OOS Phase 2 (CONTEXT.md и REQUIREMENTS не требуют визуализации статусов в UI; CHAT-03 закрыт через факт streaming delta). Phase 3 «Streaming stages в UI» (Plan 3.1).
</risks>
